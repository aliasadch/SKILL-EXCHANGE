# =====================================================
# APACHE CONFIGURATION
# =====================================================

# Enable rewrite engine
RewriteEngine On

# Set base directory (change this to your folder name)
RewriteBase /SKILL-EXCHANGE/

# Redirect trailing slashes
RewriteCond %{REQUEST_FILENAME} !-d
RewriteRule ^(.*)/$ /$1 [L,R=301]

# API Routing
RewriteRule ^api/auth/(.*)$ api/auth.php?action=$1 [QSA,L]
RewriteRule ^api/users/(.*)$ api/users.php?action=$1 [QSA,L]
RewriteRule ^api/skills/(.*)$ api/skills.php?action=$1 [QSA,L]
RewriteRule ^api/matches/(.*)$ api/matches.php?action=$1 [QSA,L]
RewriteRule ^api/sessions/(.*)$ api/sessions.php?action=$1 [QSA,L]
RewriteRule ^api/messages/(.*)$ api/messages.php?action=$1 [QSA,L]
RewriteRule ^api/reviews/(.*)$ api/reviews.php?action=$1 [QSA,L]
RewriteRule ^api/progress/(.*)$ api/progress.php?action=$1 [QSA,L]
RewriteRule ^api/admin/(.*)$ api/admin.php?action=$1 [QSA,L]

# Prevent access to sensitive files
<FilesMatch "^\.">
    Order allow,deny
    Deny from all
</FilesMatch>

<FilesMatch "^(config|includes|models)">
    Order allow,deny
    Deny from all
</FilesMatch>

# Set default charset
AddDefaultCharset UTF-8

# Enable compression
<IfModule mod_deflate.c>
    AddOutputFilterByType DEFLATE text/html text/plain text/xml text/css text/javascript application/javascript application/json
</IfModule>

# Cache control for assets
<IfModule mod_expires.c>
    ExpiresActive On
    ExpiresByType image/jpg "access plus 1 year"
    ExpiresByType image/jpeg "access plus 1 year"
    ExpiresByType image/gif "access plus 1 year"
    ExpiresByType image/png "access plus 1 year"
    ExpiresByType text/css "access plus 1 month"
    ExpiresByType application/javascript "access plus 1 month"
    ExpiresByType text/javascript "access plus 1 month"
</IfModule>

# Error documents
ErrorDocument 400 /SKILL-EXCHANGE/error/400.html
ErrorDocument 401 /SKILL-EXCHANGE/error/401.html
ErrorDocument 403 /SKILL-EXCHANGE/error/403.html
ErrorDocument 404 /SKILL-EXCHANGE/error/404.html
ErrorDocument 500 /SKILL-EXCHANGE/error/500.html

# PHP settings
php_value upload_max_filesize 10M
php_value post_max_size 10M
php_value max_execution_time 300
php_value max_input_time 300
